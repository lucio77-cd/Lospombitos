// ============================================================ // mercado-api.js — Dados de Mercado em Tempo Real // MELHORADO — sem alterar estrutura original // ============================================================ const MercadoAPI = { // ---------------------------------------------------------- // CONFIG // ---------------------------------------------------------- PROXY: 'https://corsproxy.io/?', PROXY_FALLBACK: [ 'https://corsproxy.io/?', 'https://api.allorigins.win/raw?url=' ], _cache: {}, CACHE_TIME: 1000 * 15, // 15 segundos TIMEOUT: 12000, RETRY: 2, // ---------------------------------------------------------- // REQUEST BASE // ---------------------------------------------------------- async request(url, tentativa = 0) { // CACHE const cached = this._cache[url]; if ( cached && (Date.now() - cached.timestamp < this.CACHE_TIME) ) { return cached.data; } let ultimoErro = null; // TESTA TODOS PROXIES for (const proxy of this.PROXY_FALLBACK) { try { const controller = new AbortController(); const timeout = setTimeout(() => { controller.abort(); }, this.TIMEOUT); const response = await fetch( proxy + encodeURIComponent(url), { method: 'GET', headers: { 'Accept': 'application/json', 'Cache-Control': 'no-cache' }, signal: controller.signal } ); clearTimeout(timeout); if (!response.ok) { throw new Error(`HTTP ${response.status}`); } const data = await response.json(); // SALVA CACHE this._cache[url] = { timestamp: Date.now(), data }; return data; } catch (erro) { ultimoErro = erro; console.warn( '[MercadoAPI]', 'Proxy falhou:', proxy, erro.message ); } } // RETRY if (tentativa < this.RETRY) { console.warn( '[MercadoAPI]', 'Tentando novamente...', tentativa + 1 ); await new Promise(r => setTimeout(r, 1000)); return this.request(url, tentativa + 1); } throw ultimoErro || new Error('Falha desconhecida'); }, // ---------------------------------------------------------- // NORMALIZA NÚMERO // ---------------------------------------------------------- numero(valor) { const n = Number(valor); return Number.isFinite(n) ? n : 0; }, // ---------------------------------------------------------- // FORMATA TICKER // ---------------------------------------------------------- ticker(ticker = '') { ticker = ticker .toUpperCase() .trim() .replace(/\s/g, ''); // adiciona .SA automaticamente if ( /^[A-Z]{4}\d{1,2}$/.test(ticker) && !ticker.includes('.SA') ) { ticker += '.SA'; } return ticker; }, // ---------------------------------------------------------- // COTAÇÃO // ---------------------------------------------------------- async cotacao(ticker) { try { ticker = this.ticker(ticker); const url = `https://query1.finance.yahoo.com/v8/finance/chart/${ticker}`; const data = await this.request(url); const result = data?.chart?.result?.[0]; if (!result) { throw new Error('Ticker inválido'); } const meta = result.meta || {}; return { ticker, nome: meta.longName || meta.shortName || ticker, preco: this.numero( meta.regularMarketPrice ), variacao: this.numero( meta.regularMarketChangePercent ), variacao_valor: this.numero( meta.regularMarketChange ), abertura: this.numero( meta.regularMarketOpen ), maxima: this.numero( meta.regularMarketDayHigh ), minima: this.numero( meta.regularMarketDayLow ), fechamento: this.numero( meta.previousClose ), volume: this.numero( meta.regularMarketVolume ), moeda: meta.currency || 'BRL', mercado: meta.exchangeName || 'B3', atualizado: meta.regularMarketTime ? new Date( meta.regularMarketTime * 1000 ) : new Date() }; } catch (erro) { console.error( '[MercadoAPI]', 'Erro cotação:', ticker, erro ); return { erro: true, ticker, preco: 0, variacao: 0, mensagem: erro.message }; } }, // ---------------------------------------------------------- // MÚLTIPLOS TICKERS // ---------------------------------------------------------- async cotacoes(lista = []) { const promessas = lista.map(t => this.cotacao(t) ); const resultados = await Promise.allSettled(promessas); return resultados.map(r => r.status === 'fulfilled' ? r.value : { erro: true } ); }, // ---------------------------------------------------------- // HISTÓRICO // ---------------------------------------------------------- async historico( ticker, intervalo = '1d', periodo = '1mo' ) { try { ticker = this.ticker(ticker); const url = `https://query1.finance.yahoo.com/v8/finance/chart/${ticker}` + `?interval=${intervalo}` + `&range=${periodo}`; const data = await this.request(url); const result = data?.chart?.result?.[0]; if (!result) { throw new Error('Sem histórico'); } const timestamps = result.timestamp || []; const quote = result.indicators?.quote?.[0] || {}; return timestamps.map((ts, i) => ({ data: new Date(ts * 1000), open: this.numero( quote.open?.[i] ), high: this.numero( quote.high?.[i] ), low: this.numero( quote.low?.[i] ), close: this.numero( quote.close?.[i] ), volume: this.numero( quote.volume?.[i] ) })); } catch (erro) { console.error( '[MercadoAPI]', 'Erro histórico:', erro ); return []; } } }; // ------------------------------------------------------------ // PAUSA UPDATE QUANDO ABA OCULTA // ------------------------------------------------------------ window.addEventListener( 'visibilitychange', () => { if (document.hidden) { console.log( '[MercadoAPI]', 'Atualização pausada' ); } else { console.log( '[MercadoAPI]', 'Atualização retomada' ); } } );
